npm 源管理的方式 #

推荐的方式 -- nrm 管理 #

bash
npm i nrm -g
nrm list # 查看源 前面有 * 的是正在使用的
nrm use registry-name # 切换源

nrm test registry-name # 测试
nrm add registry-name url # 添加新的源
nrm del registry-name # 删除源

nrm set-auth -u username -p password registry-name # 对源添加认证信息
nrm publish # 发布 npm
