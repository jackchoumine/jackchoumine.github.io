  1. 配置别名和扩展名

resolve函数：

js
const path = require('path')
const resolve = dir => path.resolve(__dirname, dir)
1
2

第一种方式：

js
chainWebpack: config => {
  config.resolve.alias
    .set('@', resolve('src/'))
    .set('@com', resolve('src/components'))
    .set('views', resolve('src/views'))
  config.resolve.extensions
    .add('.js')
    .add('.vue')
    .add('.json')
}
